import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import base64
from io import BytesIO

# -----------------------------
# Load synthetic dataset
# -----------------------------
data = {
    "employee_id": [f"EMP{i:03d}" for i in range(1, 101)],
    "department": ["Operations", "R&D", "Sales", "IT", "Marketing"] * 20,
    "region": ["Africa", "North America", "Latin America", "Asia Pacific", "Europe"] * 20,
    "performance_score": [round(x, 2) for x in list(range(50, 150))[:100]],
    "years_experience": [i % 15 + 1 for i in range(100)],
    "satisfaction_rating": [round(3 + (i % 3) + 0.5, 1) for i in range(100)],
}
df = pd.DataFrame(data)

# -----------------------------
# Frequency count for Marketing
# -----------------------------
marketing_count = (df["department"] == "Marketing").sum()
print("Marketing frequency:", marketing_count)

# -----------------------------
# Create histogram
# -----------------------------
plt.figure(figsize=(8, 6))
sns.set_style("whitegrid")
sns.histplot(df["department"], discrete=True, color="skyblue", edgecolor="black")
plt.title("Department Distribution of Employees")
plt.xlabel("Department")
plt.ylabel("Count")

# Save image as base64
buf = BytesIO()
plt.savefig(buf, format="png", dpi=100, bbox_inches="tight")
